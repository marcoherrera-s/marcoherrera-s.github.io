@taylorize function three_body_restricted_param!(du::Vector{T}, u::Vector{T}, p, t) where T
    x, y, v_x, v_y, Œº = u

    x_plus_mu = x + Œº
    x_minus_one_plus_mu = x - 1 + Œº
    one_minus_mu = 1 - Œº  

    r1 = sqrt(x_plus_mu^2 + y^2)
    r2 = sqrt(x_minus_one_plus_mu^2 + y^2)

    r1_cubed = r1^3
    r2_cubed = r2^3

    du[1] = v_x
    du[2] = v_y
    du[3] = (2v_y + x) - (one_minus_mu * x_plus_mu / r1_cubed + Œº * x_minus_one_plus_mu / r2_cubed)
    du[4] = (-2v_x + y) - (one_minus_mu * y / r1_cubed + Œº * y / r2_cubed)
    du[5] = zero(Œº)
end

Estas ecuaciones de movimiento representan un sistema de tres cuerpos en el que una de las masas es significativamente menor que las otras dos (problema de tres cuerpos restringido). La funci√≥n three_body_restricted_param! define c√≥mo evoluciona el sistema con el tiempo, y el uso de @taylorize permite calcular autom√°ticamente las derivadas necesarias para el Jet Transport.
